Пример нерекурсивной функции расчёта факториала числа: function fact1(n) { var f=1; for ( var i=2; i<=n; i++ ) f*=i; return f; } console.log( fact1(5) ); 120 Пример рекурсивной функции расчёта факториала числа: function fact2(n) { if ( n<=1 ) /* условие останова рекурсии */ return 1; return n*fact2(n-1); /* рекурсия */ } console.log( fact2(5) ); 120 (факториал числа N — это произведение всех целых чисел от 1 до N)
